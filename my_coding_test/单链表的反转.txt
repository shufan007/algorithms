struct NodeList {
  NodeList * next;
  int data;
};

// 2、实现单链表的反转 例如: A -> B -> C -> D -> NULL 翻转后的结果是 D -> C -> B -> A -> NULL

NodeList* ListReverse(NodeList* pHead)
{
  NodeList *p,*pNext,*pPrev;
  
  p = pHead;
  pPrev = NULL;
  while(p!=NULL)
  {
    pNext = p->next;
    p->next = pPrev;
    pPrev = p;
	p = pNext;
  }
  return p;
}
